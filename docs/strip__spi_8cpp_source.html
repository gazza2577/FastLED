28led_strip_handle_t configure_led(int pin, uint32_t led_count, led_model_t led_model, spi_host_device_t spi_bus, bool with_dma)
29{
30 // LED strip general initialization, according to your led board design
31 led_strip_config_t strip_config = {
32 .strip_gpio_num = pin, // The GPIO that connected to the LED strip's data line
33 .max_leds = led_count, // The number of LEDs in the strip,
34 .led_model = led_model, // LED strip model
35 // set the color order of the strip: GRB
36 .color_component_format = {
37 .format = {
38 .r_pos = 0, // red is the second byte in the color data
39 .g_pos = 1, // green is the first byte in the color data
40 .b_pos = 2, // blue is the third byte in the color data
41 .num_components = 3, // total 3 color components
42 },
43 },
44 .flags = {
45 .invert_out = false, // don't invert the output signal
46 }
47 };
48
49 // LED strip backend configuration: SPI
50 led_strip_spi_config_t spi_config = {
51 .clk_src = SPI_CLK_SRC_DEFAULT, // different clock source can lead to different power consumption
52 .spi_bus = spi_bus, // SPI bus ID
53 .flags = {
54 .with_dma = with_dma, // Using DMA can improve performance and help drive more LEDs
55 }
56 };
57
58 // LED Strip object handle
59 led_strip_handle_t led_strip;
60 ESP_ERROR_CHECK(led_strip_new_spi_device(&strip_config, &spi_config, &led_strip));
61 ESP_LOGI(TAG, "Created LED strip object with SPI backend");
62 return led_strip;
63}

65struct SpiHostUsed {
66 spi_host_device_t spi_host;
67 bool used;
68};

70static SpiHostUsed gSpiHostUsed[] = {
71 {SPI2_HOST, false}, // in order of preference
72#if SOC_SPI_PERIPH_NUM > 2
73 {SPI3_HOST, false},
74#endif
75 {SPI1_HOST, false},
76};

78static spi_host_device_t getNextAvailableSpiHost()
79{
80 for (int i = 0; i < sizeof(gSpiHostUsed) / sizeof(gSpiHostUsed[0]); i++)
81 {
82 if (!gSpiHostUsed[i].used)
83 {
84 gSpiHostUsed[i].used = true;
85 return gSpiHostUsed[i].spi_host;
86 }
87 }
88 ESP_ERROR_CHECK(ESP_ERR_NOT_FOUND);
89 return SPI_HOST_MAX;
90}
91
92
93static void releaseSpiHost(spi_host_device_t spi_host)
94{
95 for (int i = 0; i < sizeof(gSpiHostUsed) / sizeof(gSpiHostUsed[0]); i++)
96 {
97 if (gSpiHostUsed[i].spi_host == spi_host)
98 {
99 gSpiHostUsed[i].used = false;
100 return;
101 }
102 }
103 ESP_ERROR_CHECK(ESP_ERR_NOT_FOUND);
104}

107class SpiStripWs2812 : public ISpiStripWs2812 {
108public:
109 SpiStripWs2812(int pin, uint32_t led_count, ISpiStripWs2812::SpiHostMode spi_bus_mode, ISpiStripWs2812::DmaMode dma_mode = DMA_AUTO)
110 : mIsRgbw(false), // SPI implementation currently only supports RGB
111 mLedCount(led_count)
112 {
113 switch (spi_bus_mode) {
114 case ISpiStripWs2812::SPI_HOST_MODE_AUTO:
115 mSpiHost = getNextAvailableSpiHost();
116 break;
117 case ISpiStripWs2812::SPI_HOST_MODE_1:
118 mSpiHost = SPI1_HOST;
119 break;
120 case ISpiStripWs2812::SPI_HOST_MODE_2:
121 mSpiHost = SPI2_HOST;
122 break;
123#if SOC_SPI_PERIPH_NUM > 2
124 case ISpiStripWs2812::SPI_HOST_MODE_3:
125 mSpiHost = SPI3_HOST;
126 break;
127#endif
128 default:
129 ESP_ERROR_CHECK(ESP_ERR_NOT_SUPPORTED);
130 }
131
132 bool with_dma = dma_mode == ISpiStripWs2812::DMA_ENABLED || dma_mode == ISpiStripWs2812::DMA_AUTO;
133 led_strip_handle_t led_strip = configure_led(pin, led_count, LED_MODEL_WS2812, mSpiHost, with_dma);
134 mStrip = led_strip;
135 }
136
137 ~SpiStripWs2812() override
138 {
139 waitDone();
140 led_strip_del(mStrip);
141 releaseSpiHost(mSpiHost);
142 mStrip = nullptr;
143 }
144
145 void setPixel(uint32_t index, uint8_t red, uint8_t green, uint8_t blue) override
146 {
147 ESP_ERROR_CHECK(led_strip_set_pixel(mStrip, index, red, green, blue));
148 }
149
150 void drawAsync() override
151 {
152 if (mDrawIssued)

154 waitDone();

162 if (!mDrawIssued)

164 return;

166 ESP_ERROR_CHECK(led_strip_refresh_wait_done(mStrip));
167 mDrawIssued = false;
168 }
169
170 bool isDrawing() override
171 {
172 return mDrawIssued;
173 }
174
175 void fill(uint8_t red, uint8_t green, uint8_t blue) {
176 for (int i = 0; i < mLedCount; i++)
177 {
178 setPixel(i, red, green, blue);
179 }
180 }
181
182 void clear()
183 {
184 ESP_ERROR_CHECK(led_strip_clear(mStrip));
185 }
186
187 void fill_color(uint8_t red, uint8_t green, uint8_t blue)
188 {
189 for (int i = 0; i < mLedCount; i++)
190 {
191 setPixel(i, red, green, blue);
192 }
193 }
194
195 OutputIterator outputIterator() override
196 {
197 return OutputIterator(this, mLedCount);
198 }
199
200 uint32_t numPixels() override
201 {
202 return mLedCount;
203 }
204
205private:
206 spi_host_device_t mSpiHost = SPI2_HOST;
207 led_strip_handle_t mStrip;
208 bool mDrawIssued = false;
209 bool mIsRgbw;
210 uint32_t mLedCount = 0;
211};

226void ISpiStripWs2812::OutputIterator::flush() {
227 mStrip->setPixel(mPosition, mRed, mGreen, mBlue);
228 mRed = mGreen = mBlue = 0;
229}

231void ISpiStripWs2812::OutputIterator::operator()(uint8_t value) {
232 switch (mWritten) {
233 case 0:
234 mRed = value;
235 break;
236 case 1:
237 mGreen = value;
238 break;
239 case 2:
240 mBlue = value;
241 break;
242 }
243 mWritten++;
244 if (mWritten == 3) {
245 flush();
246 mWritten = 0;
247 mPosition++;
248 }
249}

251void ISpiStripWs2812::OutputIterator::finish() {
252 if (mWritten) {
253 flush();
254 }
