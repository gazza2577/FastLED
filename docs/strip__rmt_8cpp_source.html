20// 10MHz resolution, 1 tick = 0.1us (led strip needs a high resolution)
21#define LED_STRIP_RMT_RES_HZ (10 * 1000 * 1000)

23led_strip_handle_t configure_led_with_timings(int pin, uint32_t led_count, bool is_rgbw, uint32_t t0h, uint32_t t0l, uint32_t t1h, uint32_t t1l, uint32_t reset, bool with_dma, uint8_t interrupt_priority)
24{
25 led_strip_encoder_timings_t timings = {
26 .t0h = t0h,
27 .t1h = t1h,
28 .t0l = t0l,
29 .t1l = t1l,
30 .reset = reset};
31
32 // is always going to fail, so it's disabled for now.
33 uint32_t memory_block_symbols = with_dma ? 1024 : 0;
34 led_color_component_format_t color_component_format =
35 is_rgbw ? LED_STRIP_COLOR_COMPONENT_FMT_RGBW : LED_STRIP_COLOR_COMPONENT_FMT_RGB;
36
37 // LED strip general initialization, according to your led board design
38 led_strip_config_t strip_config = {
39 .strip_gpio_num = pin, // The GPIO that connected to the LED strip's data line
40 .max_leds = led_count, // The number of LEDs in the strip,
41 .led_model = LED_MODEL_WS2812, // LED strip model
42 .color_component_format = color_component_format, // The color order of the strip: GRB
43 .flags = {
44 .invert_out = false, // don't invert the output signal
45 },
46 .timings = timings
47 };
48
49 // LED strip backend configuration: RMT
50 led_strip_rmt_config_t rmt_config = {
51 .clk_src = RMT_CLK_SRC_DEFAULT, // different clock source can lead to different power consumption
52 .resolution_hz = LED_STRIP_RMT_RES_HZ, // RMT counter clock frequency
53 .mem_block_symbols = memory_block_symbols, // the memory size of each RMT channel, in words (4 bytes)
54 .flags = {
55 .with_dma = false, // DMA feature is available on chips like ESP32-S3/P4
56 },
57 .interrupt_priority = interrupt_priority, // RMT interrupt priority
58 };
59
60 // LED Strip object handle
61 led_strip_handle_t led_strip;
62 ESP_ERROR_CHECK(led_strip_new_rmt_device(&strip_config, &rmt_config, &led_strip));
63 ESP_LOGI(TAG, "Created LED strip object with RMT backend");
64 return led_strip;
65
66 return nullptr;
67}

70class RmtStrip : public IRmtStrip

72public:
73 RmtStrip(int pin, uint32_t led_count, bool is_rgbw, uint32_t th0, uint32_t tl0, uint32_t th1, uint32_t tl1, uint32_t reset, IRmtStrip::DmaMode dma_mode, uint8_t interrupt_priority)
74 : mIsRgbw(is_rgbw), mLedCount(led_count)
75 {

81 ~RmtStrip() override
82 {
83 waitDone();
84 led_strip_del(mStrip);
85 mStrip = nullptr;
86 }

88 void setPixel(uint32_t index, uint8_t red, uint8_t green, uint8_t blue) override
89 {
90 ESP_ERROR_CHECK(mIsRgbw ? ESP_ERR_INVALID_ARG : ESP_OK);
91 ESP_ERROR_CHECK(led_strip_set_pixel(mStrip, index, red, green, blue));
92 }
93
94 void setPixelRGBW(uint32_t index, uint8_t red, uint8_t green, uint8_t blue, uint8_t white) override
95 {
96 ESP_ERROR_CHECK(mIsRgbw ? ESP_OK : ESP_ERR_INVALID_ARG);
97 ESP_ERROR_CHECK(led_strip_set_pixel_rgbw(mStrip, index, red, green, blue, white));
98 }
99
100 void drawAsync() override
101 {
102 if (mDrawIssued)
103 {
104 waitDone();
105 }
106 ESP_ERROR_CHECK(led_strip_refresh_async(mStrip));
107 mDrawIssued = true;
108 }
109
110 void waitDone() override
111 {
112 if (!mDrawIssued)
113 {
114 // ESP_LOGE(TAG, "No draw issued, skipping wait");
115 return;
116 }
117 ESP_ERROR_CHECK(led_strip_refresh_wait_done(mStrip));
118 mDrawIssued = false;
119 }
120
121 bool isDrawing() override
122 {
123 return mDrawIssued;
124 }
125
126 void clear()
127 {
128 ESP_ERROR_CHECK(led_strip_clear(mStrip));
129 }
130
131 void fill(uint8_t red, uint8_t green, uint8_t blue) override {
132 for (int i = 0; i < mLedCount; i++)
133 {
134 setPixel(i, red, green, blue);
135 }
136 }
137
138 void fillRGBW(uint8_t red, uint8_t green, uint8_t blue, uint8_t white) override {
139 for (int i = 0; i < mLedCount; i++)
140 {
141 setPixelRGBW(i, red, green, blue, white);
142 }
143 }
144
145 uint32_t numPixels() override
146 {
147 return mLedCount;
148 }
149
150private:
151 led_strip_handle_t mStrip;
152 bool mDrawIssued = false;
153 bool mIsRgbw;
154 uint32_t mLedCount;
155};
